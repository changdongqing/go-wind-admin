#### JSON 字段强类型 + Wrapper（遵循 CLAUDE.md "Step 13"）

> ⚠️ 直接把 `*thingmodelV1.FeatureSpec`/`*thingmodelV1.FeatureOverrideSpec` 喂给 `field.JSON` 是危险的：
> - `FeatureSpec` 是 `oneof`（PropertySpec/EventSpec/ServiceSpec/RelationSpec），Go 的 `encoding/json` 不能正确 round-trip oneof；
> - 写入侥幸成功、读取必崩 `cannot unmarshal ... into isXxx_Yyy`。

**强制做法**：

```go
// internal/data/ent/converter/feature_spec_field.go
package converter

import (
    "encoding/json"

    "google.golang.org/protobuf/encoding/protojson"

    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"
)

// FeatureSpecField 包装 *thingmodelV1.FeatureSpec 走 protojson
type FeatureSpecField struct {
    Spec *thingmodelV1.FeatureSpec
}

func (f *FeatureSpecField) MarshalJSON() ([]byte, error) {
    if f == nil || f.Spec == nil { return []byte("null"), nil }
    return protojson.Marshal(f.Spec)
}
func (f *FeatureSpecField) UnmarshalJSON(b []byte) error {
    if len(b) == 0 || string(b) == "null" { return nil }
    f.Spec = &thingmodelV1.FeatureSpec{}
    return protojson.Unmarshal(b, f.Spec)
}

// 同理：FeatureOverrideSpecField
type FeatureOverrideSpecField struct {
    Spec *thingmodelV1.FeatureOverrideSpec
}
// MarshalJSON / UnmarshalJSON 同上（无 oneof，但保持风格统一）
```

Schema 中使用：

```go
field.JSON("feature_snapshot", &converter.FeatureSpecField{}).
    Comment("完整 spec 快照 / Full feature spec snapshot"),
field.JSON("override_spec", &converter.FeatureOverrideSpecField{}).
    Optional().Nillable().
    Comment("稀疏覆写 / Sparse override"),
```

Repo 层的 copier mapper 注册 TypeConverter：

```go
copier.RegisterTypeConverter(
    &converter.FeatureSpecField{}, (*thingmodelV1.FeatureSpec)(nil),
    func(src interface{}) (interface{}, error) {
        return src.(*converter.FeatureSpecField).Spec, nil
    },
)
// 反向同理
```

## 3. 覆写白名单校验器（核心业务规则）

```go
// internal/biz/thingmodel_override_validator.go
package biz

import (
    "google.golang.org/protobuf/reflect/protoreflect"

    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"
    thingmodelErr "go-wind-admin/api/gen/go/thingmodel/service/v1"
)

// ValidateOverrideSpec 校验覆写 spec 仅包含白名单字段
// 实现：FeatureOverrideSpec 本身就是白名单结构（proto 层就限定了），但仍要做：
//   1. unit 引用的 unitId 必须存在；
//   2. constraints 数值类型与目标特征 dataType 兼容；
//   3. defaultValue 类型与 dataType 一致。
func (v *OverrideValidator) ValidateOverrideSpec(
    targetSpec *thingmodelV1.FeatureSpec,
    override *thingmodelV1.FeatureOverrideSpec,
) error {
    if override == nil { return nil }

    p := targetSpec.GetProperty()
    if p == nil {
        // 非 property 类型禁止覆写 constraints/unit/defaultValue
        if override.Constraints != nil || override.Unit != nil || override.DefaultValue != nil {
            return thingmodelErr.ErrorTmPfOverrideInvalid("non-property cannot override constraints/unit/defaultValue")
        }
        return nil
    }

    // constraints 兼容性：min/max/step 必须为数值类型且 min <= max
    if override.Constraints != nil {
        if override.Constraints.Max < override.Constraints.Min {
            return thingmodelErr.ErrorTmPfOverrideInvalid("constraints.max < min")
        }
    }

    // unit 引用必须存在
    if override.Unit != nil && override.Unit.UnitId > 0 {
        if exists, _ := v.unitRepo.Exists(ctx, override.Unit.UnitId); !exists {
            return thingmodelErr.ErrorTmPfOverrideInvalid("unit not found")
        }
    }

    return nil
}
```

**proto 层"结构性白名单"** = `FeatureOverrideSpec` 消息**只包含**白名单 6 个字段，根本无法在线上传 `dataType` 等黑名单字段。这是"靠 schema 收口"而非"靠 if 收口"的设计。

## 4. PullFromDefault Service 实现要点

```go
// internal/biz/thingmodel_product_feature.go
func (uc *ProductFeatureUseCase) PullFromDefault(
    ctx context.Context, req *bizdto.PullFromDefaultReq,
) ([]*bizdto.ProductFeatureBO, []*bizdto.PullSkipped, error) {

    // 1. 读产品
    product, err := uc.productRepo.Get(ctx, req.ProductId)
    if err != nil { return nil, nil, err }
    if product.Status == "PUBLISHED" {
        return nil, nil, thingmodelErr.ErrorTmProductStatusNotDraft("cannot pull when published")
    }

    // 2. 取分类默认条目列表
    cdfs, err := uc.cdfRepo.ListByCategory(ctx, product.CategoryId, req.DefaultFeatureIds)
    if err != nil { return nil, nil, err }

    // 3. 现有产品特征 feature_id 集合（用于冲突判定）
    existing, err := uc.pfRepo.ListByProductFeatureIds(ctx, product.Id, collectFeatureIds(cdfs))
    if err != nil { return nil, nil, err }
    existingByFeatureId := indexByFeatureId(existing)

    // 4. 事务批量创建
    var created []*bizdto.ProductFeatureBO
    var skipped []*bizdto.PullSkipped

    err = uc.txm.RunInTx(ctx, func(txCtx context.Context) error {
        for _, cdf := range cdfs {
            // 冲突处理
            if pf, dup := existingByFeatureId[cdf.FeatureId]; dup {
                switch req.OnConflict {
                case "REPLACE":
                    if err := uc.pfRepo.Delete(txCtx, pf.Id); err != nil { return err }
                case "SKIP", "":
                    skipped = append(skipped, &bizdto.PullSkipped{
                        DefaultFeatureId: cdf.Id, Reason: "duplicate",
                    })
                    continue
                }
            }

            // 取全局特征 spec
            feat, err := uc.featureRepo.Get(txCtx, cdf.FeatureId)
            if err != nil { return err }
            if !feat.IsEnabled {
                skipped = append(skipped, &bizdto.PullSkipped{
                    DefaultFeatureId: cdf.Id, Reason: "feature_disabled",
                })
                continue
            }

            // 合并默认模型层 override 到 snapshot
            snapshot := MergeOverride(feat.Spec, cdf.OverrideSpec)

            // 构造 ProductFeature
            pf := &bizdto.ProductFeatureBO{
                ProductId:      product.Id,
                Source:         "DEFAULT",
                RefFeatureId:   feat.Id,
                FeatureType:    feat.FeatureType,
                Code:           feat.Code,
                Identifier:     feat.Identifier,
                Name:           feat.Name,
                NameEn:         feat.NameEn,
                Description:    feat.Description,
                FeatureSnapshot: snapshot,
                OverrideSpec:    nil,  // 产品层暂未覆写
                // 冗余特化列同步
                DataType: feat.DataType, AccessMode: feat.AccessMode,
                EventLevel: feat.EventLevel, CallMode: feat.CallMode,
                RelationType: feat.RelationType,
                SortOrder: cdf.SortOrder, IsEnabled: true,
                TenantId: product.TenantId,
            }

            createdBO, err := uc.pfRepo.Create(txCtx, pf)
            if err != nil { return err }
            created = append(created, createdBO)

            // ⚠️ source=DEFAULT 不动 reference_count（由 cdf 一侧维护过）
        }
        return nil
    })

    return created, skipped, err
}
```

## 5. effective_spec 计算

```go
// internal/biz/thingmodel_effective_spec.go
package biz

import "google.golang.org/protobuf/proto"

// EffectiveSpec 计算运行时有效 spec = snapshot deepmerged with override
func EffectiveSpec(snapshot *thingmodelV1.FeatureSpec, override *thingmodelV1.FeatureOverrideSpec) *thingmodelV1.FeatureSpec {
    if override == nil { return snapshot }
    out := proto.Clone(snapshot).(*thingmodelV1.FeatureSpec)

    switch s := out.Spec.(type) {
    case *thingmodelV1.FeatureSpec_Property:
        if override.Constraints != nil { s.Property.Constraints = override.Constraints }
        if override.Unit != nil        { s.Property.Unit = override.Unit }
        if override.DefaultValue != nil{ s.Property.DefaultValue = override.DefaultValue }
    case *thingmodelV1.FeatureSpec_Event:
        // event 仅 displayName/description 走列字段，不在 spec 内
    case *thingmodelV1.FeatureSpec_Service:
        // service 内 inputParams.required 可走 override.required（针对单参，按 key 匹配）
        // 实际实现按业务需要定 — 本期保守只处理 property
    case *thingmodelV1.FeatureSpec_Relation:
        // 暂不支持
    }
    return out
}
```

> **`GetProductFeature` 返回时**：response 同时返回 `feature_snapshot`（原始快照）+ `override_spec`（稀疏覆写）+ 后端拼好的 `effective_spec`（运行时 = 前者 merge 后者）。前端展示用 effective，编辑时定位到 override。

## 6. reference_count 维护

```go
// internal/biz/thingmodel_category_default_feature.go
func (uc *CategoryDefaultFeatureUseCase) Create(ctx context.Context, bo *bizdto.CategoryDefaultFeatureBO) (*bizdto.CategoryDefaultFeatureBO, error) {
    return uc.txm.RunInTxResult(ctx, func(txCtx context.Context) (*bizdto.CategoryDefaultFeatureBO, error) {
        // 1. 校验 category.level == 4
        cat, err := uc.categoryRepo.Get(txCtx, bo.CategoryId)
        if err != nil { return nil, err }
        if cat.Level != 4 {
            return nil, thingmodelErr.ErrorTmCatDefaultFeatureCategoryNotLeaf(...)
        }

        // 2. 校验 override_spec
        feat, err := uc.featureRepo.Get(txCtx, bo.FeatureId)
        if err != nil { return nil, err }
        if !feat.IsEnabled {
            return nil, thingmodelErr.ErrorTmCatDefaultFeatureFeatureDisabled(...)
        }
        if err := uc.overrideValidator.ValidateOverrideSpec(feat.Spec, bo.OverrideSpec); err != nil {
            return nil, err
        }

        // 3. 创建关联行
        created, err := uc.cdfRepo.Create(txCtx, bo)
        if err != nil { return nil, err }

        // 4. 维护 reference_count
        if err := uc.featureRepo.IncRefCount(txCtx, bo.FeatureId, 1); err != nil { return nil, err }
        if unitId := pickUnitId(feat.Spec, bo.OverrideSpec); unitId > 0 {
            if err := uc.unitRepo.IncRefCount(txCtx, unitId, 1); err != nil { return nil, err }
        }
        return created, nil
    })
}
```

**删除**对称：先取行 → 删行 → feature ref --，unit ref --。
**修改 override.unit**（同事务）：旧 unit -1，新 unit +1。

## 7. CreateProduct + PublishProduct

```go
// internal/biz/thingmodel_product.go
func (uc *ProductUseCase) Create(ctx context.Context, bo *bizdto.ProductBO) (*bizdto.ProductBO, error) {
    // 1. 校验 category.level == 4
    cat, err := uc.categoryRepo.Get(ctx, bo.CategoryId)
    if err != nil { return nil, err }
    if cat.Level != 4 {
        return nil, thingmodelErr.ErrorTmProductCategoryNotLeaf(...)
    }
    // 2. 默认 status=DRAFT
    if bo.Status == "" { bo.Status = "DRAFT" }
    // 3. 唯一性由 DB 兜底（捕获 unique violation → 转 TM_PRODUCT_CODE_DUPLICATE / NAME_DUPLICATE）
    return uc.productRepo.Create(ctx, bo)
}

func (uc *ProductUseCase) Publish(ctx context.Context, id uint32) error {
    p, err := uc.productRepo.Get(ctx, id)
    if err != nil { return err }
    if p.Status == "PUBLISHED" { return nil }
    return uc.productRepo.UpdateStatus(ctx, id, "PUBLISHED")
}
```
