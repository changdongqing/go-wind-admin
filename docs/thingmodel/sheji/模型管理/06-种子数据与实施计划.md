# 06 · 种子数据与实施计划

> 本章给出：本期最小可演示种子（暖通冷水机组一条完整链路）、Go 种子注入程序的落点、分阶段任务、验收标准。

---

## 1. 种子设计原则

| 原则 | 说明 |
|------|------|
| **最小可演示**（MVD） | 只做"暖通冷水机组"一条端到端链路：分类 → 默认模型 → 1 个示范产品 |
| **零侵入已落地种子** | 不修改 `分类管理 / 特征管理 / 单位管理` 已有种子；只在它们之上追加默认模型与产品 |
| **租户隔离** | 全部种子 `tenant_id = 0`（系统全局），用户租户复制时可作为模板 |
| **idempotent** | 重跑种子不出错；按 `(tenant_id, category_id, feature_id)` upsert |
| **延迟执行** | 种子在 `migration` 完成 & 特征/单位/分类种子都执行完之后再跑 |

### 3.2 程序骨架

```go
// internal/seed/thingmodel/model_management.go
package thingmodel

import (
    "context"
    "fmt"

    "go-wind-admin/internal/data/ent"
    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"
)

type ModelMgmtSeeder struct {
    client *ent.Client
}

func NewModelMgmtSeeder(c *ent.Client) *ModelMgmtSeeder { return &ModelMgmtSeeder{client: c} }

func (s *ModelMgmtSeeder) Run(ctx context.Context) error {
    // 1. 取目标分类
    cat, err := s.client.Category.Query().
        Where(category.KindEQ("FACILITY"), category.CodeEQ("20010100"), category.TenantIDEQ(0)).
        Only(ctx)
    if err != nil { return fmt.Errorf("seed cat: %w", err) }

    // 2. 取 10 个特征
    featureCodes := []string{
        "P-RUN-0001","P-RUN-0002","P-MEAS-0010","P-MEAS-0011","P-MEAS-0030",
        "P-RATED-0050","E-FAULT-0001","E-GEN-0001","S-CTRL-0001","R-PART-0001",
    }
    features, err := s.client.Feature.Query().
        Where(feature.CodeIn(toAnySlice(featureCodes)...), feature.TenantIDEQ(0)).
        All(ctx)
    if err != nil { return err }
    featureByCode := indexByCode(features)

    // 3. upsert 默认模型条目
    overrides := map[string]*thingmodelV1.FeatureOverrideSpec{
        "P-MEAS-0010": {Constraints: &thingmodelV1.ValueConstraints{Min: 4, Max: 15}},
        "P-MEAS-0011": {Constraints: &thingmodelV1.ValueConstraints{Min: 6, Max: 18}},
    }
    for i, code := range featureCodes {
        f := featureByCode[code]
        if f == nil { continue }
        err := s.client.CategoryDefaultFeature.Create().
            SetCategoryID(cat.ID).
            SetFeatureID(f.ID).
            SetSortOrder(uint32(i)).
            SetNillableOverrideSpec(wrapOverride(overrides[code])).
            OnConflict(
                sql.ConflictColumns("tenant_id","category_id","feature_id"),
            ).Ignore().Exec(ctx)
        if err != nil { return fmt.Errorf("seed cdf %s: %w", code, err) }
    }

    // 4. upsert 产品
    p, err := s.client.Product.Create().
        SetCode("GREE-LSBLG320").
        SetName("格力螺杆冷水机组 LSBLG320").
        SetCategoryID(cat.ID).
        SetManufacturer("格力").
        SetModelNo("LSBLG320").
        SetStatus("PUBLISHED").
        SetDescription("示范产品 — 演示从默认模型拉取与本地特征").
        OnConflict(sql.ConflictColumns("tenant_id","code")).Ignore().
        UpsertOne(ctx)
    if err != nil { return err }

    // 5. 产品特征：8 个 DEFAULT + 1 个 LOCAL（跳过 P-MEAS-0030 与 R-PART-0001）
    skipDefault := map[string]bool{"P-MEAS-0030":true, "R-PART-0001":true}
    for i, code := range featureCodes {
        if skipDefault[code] { continue }
        f := featureByCode[code]
        snapshot := mergeOverride(f.Spec, overrides[code])
        var override *thingmodelV1.FeatureOverrideSpec
        if code == "P-MEAS-0010" {
            override = &thingmodelV1.FeatureOverrideSpec{
                Constraints: &thingmodelV1.ValueConstraints{Min:5, Max:12},
            }
        }
        err := s.client.ProductFeature.Create().
            SetProductID(p.ID).
            SetSource("DEFAULT").
            SetRefFeatureID(f.ID).
            SetFeatureType(f.FeatureType).
            SetCode(f.Code).
            SetIdentifier(f.Identifier).
            SetName(f.Name).
            SetFeatureSnapshot(wrapSpec(snapshot)).
            SetNillableOverrideSpec(wrapOverride(override)).
            SetSortOrder(uint32(i)).
            // 冗余列同步
            SetNillableDataType(f.DataType).
            SetNillableAccessMode(f.AccessMode).
            SetNillableEventLevel(f.EventLevel).
            SetNillableCallMode(f.CallMode).
            SetNillableRelationType(f.RelationType).
            OnConflict(sql.ConflictColumns("product_id","code")).Ignore().Exec(ctx)
        if err != nil { return err }
    }

    // 6. LOCAL 特征：夜间静音
    localSnapshot := &thingmodelV1.FeatureSpec{
        Spec: &thingmodelV1.FeatureSpec_Property{
            Property: &thingmodelV1.PropertySpec{
                DataType:   thingmodelV1.DataType_BOOL,
                AccessMode: thingmodelV1.AccessMode_RW,
                BoolLabels: &thingmodelV1.BoolLabels{FalseLabel:"关", TrueLabel:"开"},
            },
        },
    }
    err = s.client.ProductFeature.Create().
        SetProductID(p.ID).
        SetSource("LOCAL").
        SetFeatureType("PROPERTY").
        SetCode("L-NIGHT-MUTE").
        SetIdentifier("nightMute").
        SetName("夜间静音模式").
        SetFeatureSnapshot(wrapSpec(localSnapshot)).
        SetDataType("BOOL").
        SetAccessMode("RW").
        SetSortOrder(100).
        OnConflict(sql.ConflictColumns("product_id","code")).Ignore().Exec(ctx)
    return err
}
```

### 3.3 注册到主种子流程

```go
// internal/seed/seeder.go (或现有的 seeder 注册中心)
func RunAll(ctx context.Context, c *ent.Client) error {
    seeders := []Seeder{
        // ... existing
        unit.NewSeeder(c),
        feature.NewSeeder(c),
        category.NewSeeder(c),
        thingmodel.NewModelMgmtSeeder(c),   // ← 新增，必须在前三个之后
    }
    for _, s := range seeders {
        if err := s.Run(ctx); err != nil { return err }
    }
    return nil
}
```

## 6. 风险与缓解

| 风险 | 影响 | 缓解 |
|------|------|------|
| `FeatureSpec` oneof 的 JSON round-trip 写入侥幸通过、读取崩 | 严重 | 强制走 `FeatureSpecField` wrapper（[04 §2.2](./04-后端实现设计.md)）+ Repo 测试覆盖 round-trip |
| PullFromDefault 在大默认模型（100+）下性能 | 中 | 单事务批量 INSERT；后续若慢可改为 multi-row INSERT |
| 三前端表单组件实现差异导致 UX 不一致 | 中 | 设计文档 §1 锁定形态、§5 共享组件命名；评审 UI 走样必返工 |
| 用户错把 `PUBLISHED` 当"上线"理解，期待生效 | 低 | Publish 二次确认 Modal 明确文案"冻结结构"；未来"上线给设备使用"再单独做 |
| 默认模型批量删除后 reference_count 没归零 | 中 | 单元测试覆盖 + 后台校验脚本（可选） |
