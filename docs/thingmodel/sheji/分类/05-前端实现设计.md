# 05 · 前端实现设计（v2.0）

> 仅 React 端。
> **本期采用极简形态：Tabs（切 kind）+ 单一 ProTable（按 code 升序自然展示层级）+ Drawer**。
> 不使用 Tree 组件、不使用 Splitter 分栏——code 自带层级前缀，单表排序即一目了然。

## 2. Hooks 层（`api/hooks/category.ts`）

镜像 `unit.ts`。类型从 `@/api/generated/admin/service/v1` 导入（`make ts` 后生成 `categoryservicev1_*`）。

```typescript
import { useMutation, useQuery, type UseMutationOptions, type UseQueryOptions } from '@tanstack/react-query';
import {
  type categoryservicev1_Category,
  type categoryservicev1_ListCategoryResponse,
  type categoryservicev1_GetCategoryRequest,
  type categoryservicev1_CreateCategoryRequest,
  type categoryservicev1_DeleteCategoryRequest,
} from '@/api/generated/admin/service/v1';
import { makeUpdateMask, type PaginationQuery } from '@/core/transport/rest';
import { queryClient } from '@/core';
import { apiClient } from '@/api/client';

export function useListCategories(q: PaginationQuery, o?: UseQueryOptions<categoryservicev1_ListCategoryResponse, Error>) {
  return useQuery({
    queryKey: ['listCategories', q],
    queryFn: () => apiClient.categoryService.List(q.toRawParams()),
    ...o,
  });
}
export async function fetchListCategories(params: PaginationQuery) {
  return queryClient.fetchQuery({
    queryKey: ['listCategories', params],
    queryFn: () => apiClient.categoryService.List(params.toRawParams()),
    retry: 0,
  });
}

export function useCreateCategory(o?: UseMutationOptions<{}, Error, categoryservicev1_CreateCategoryRequest>) {
  return useMutation({ mutationFn: (data) => apiClient.categoryService.Create(data), ...o });
}

export function useUpdateCategory(o?: UseMutationOptions<{}, Error, { id: number; values: Record<string, any> }>) {
  return useMutation({
    mutationFn: ({ id, values }) =>
      apiClient.categoryService.Update({
        id,
        data: { ...values },
        updateMask: makeUpdateMask(Object.keys(values ?? {})),
      }),
    ...o,
  });
}

export function useDeleteCategory(o?: UseMutationOptions<{}, Error, categoryservicev1_DeleteCategoryRequest>) {
  return useMutation({ mutationFn: (data) => apiClient.categoryService.Delete(data), ...o });
}
```

> `apiClient.categoryService` getter 需在 `src/api/client.ts` 按现有懒加载模式新增。

→ **5 个 hook**，相比 v1.0 的 9 个（含 useGetCategoryTree / useListCategoryChildren）减少近半。

## 3. 页面（`index.tsx`）

```tsx
import { useRef, useState } from 'react';
import { Tabs, Button, Popconfirm, Tag, Space, message } from 'antd';
import { ProTable, type ActionType, type ProColumns } from '@ant-design/pro-components';
import { useTranslation } from 'react-i18next';
import ContentContainer from '@/layouts/components/PageContainer/ContentContainer';
import { useProTableScrollY } from '@/hooks';
import { PaginationQuery } from '@/core/transport/rest';
import { fetchListCategories, useDeleteCategory } from '@/api/hooks';
import { useAccess } from '@/access';
import type { categoryservicev1_Category } from '@/api/generated/admin/service/v1';
import CategoryDrawer from './CategoryDrawer';
import { CATEGORY_KIND_TABS, levelColor, levelLabel } from './constants';

type Kind = 'SYSTEM' | 'SPACE' | 'FACILITY';

const CategoryManagement = () => {
  const { t } = useTranslation('category');
  const [activeKind, setActiveKind] = useState<Kind>('SYSTEM');
  const [drawer, setDrawer] = useState<{ open: boolean; record?: categoryservicev1_Category; parent?: categoryservicev1_Category }>({ open: false });
  const actionRef = useRef<ActionType>();
  const scrollY = useProTableScrollY();
  const access = useAccess();
  const { mutate: del } = useDeleteCategory({ onSuccess: () => { message.success(t('actions.deleteSuccess')); actionRef.current?.reload(); } });

  const columns: ProColumns<categoryservicev1_Category>[] = [
    {
      title: t('fields.code'),
      dataIndex: 'code',
      width: 160,
      fixed: 'left',
      // 用前缀缩进表达层级：level=1 无缩进，level=2 缩进 2 字符，依此类推
      render: (_, row) => {
        const lvl = row.level ?? 1;
        const indent = '  '.repeat(lvl - 1);   // 2× nbsp per level
        return <span style={{ fontFamily: 'monospace' }}>{indent}{row.code}</span>;
      },
    },
    { title: t('fields.name'), dataIndex: 'name', width: 240, ellipsis: true },
    { title: t('fields.nameEn'), dataIndex: 'nameEn', width: 200, hideInSearch: true, ellipsis: true },
    {
      title: t('fields.level'),
      dataIndex: 'level',
      width: 90,
      valueType: 'select',
      valueEnum: { 1: { text: t('level.1') }, 2: { text: t('level.2') }, 3: { text: t('level.3') }, 4: { text: t('level.4') } },
      render: (_, row) => <Tag color={levelColor[row.level ?? 1]}>{levelLabel(t, row.level)}</Tag>,
    },
    {
      title: t('fields.isEnabled'),
      dataIndex: 'isEnabled',
      width: 90,
      valueType: 'select',
      valueEnum: { true: { text: t('tags.enabled'), status: 'Success' }, false: { text: t('tags.disabled'), status: 'Default' } },
    },
    { title: t('fields.referenceCount'), dataIndex: 'referenceCount', width: 100, hideInSearch: true },
    { title: t('fields.sortOrder'), dataIndex: 'sortOrder', width: 90, hideInSearch: true },
    {
      title: t('common.action'),
      key: 'action',
      width: 220,
      fixed: 'right',
      hideInSearch: true,
      render: (_, row) => (
        <Space>
          {access.hasAccessByCodes(['thingmodel:category:create']) && (row.level ?? 1) < 4 && (
            <Button type="link" size="small" onClick={() => setDrawer({ open: true, parent: row })}>{t('actions.createChild')}</Button>
          )}
          {access.hasAccessByCodes(['thingmodel:category:update']) && (
            <Button type="link" size="small" onClick={() => setDrawer({ open: true, record: row })}>{t('actions.edit')}</Button>
          )}
          {access.hasAccessByCodes(['thingmodel:category:delete']) && (
            <Popconfirm title={t('tips.deleteConfirm')} onConfirm={() => del({ ids: [row.id!] })}>
              <Button type="link" size="small" danger>{t('actions.delete')}</Button>
            </Popconfirm>
          )}
        </Space>
      ),
    },
  ];

  return (
    <ContentContainer heightMode="fixed" padding="16px" bottomMargin={0}>
      <Tabs
        activeKey={activeKind}
        onChange={(k) => { setActiveKind(k as Kind); actionRef.current?.reload(); }}
        items={CATEGORY_KIND_TABS(t)}
      />
      <ProTable<categoryservicev1_Category>
        actionRef={actionRef}
        columns={columns}
        rowKey="id"
        search={{ labelWidth: 'auto' }}
        scroll={{ y: scrollY, x: 1100 }}
        request={async (params, sort, _filter) => {
          const query = new PaginationQuery({
            ...params,
            fieldValues: { kind: activeKind, ...stripPagination(params) },
            // 默认按 code 升序——天然层级序
            orderBy: sort && Object.keys(sort).length ? sort : { code: 'asc' },
          });
          const res = await fetchListCategories(query);
          return { data: res.items ?? [], total: Number(res.total ?? 0), success: true };
        }}
        toolBarRender={() =>
          access.hasAccessByCodes(['thingmodel:category:create'])
            ? [<Button key="add" type="primary" onClick={() => setDrawer({ open: true })}>{t('actions.createTopLevel')}</Button>]
            : []
        }
      />
      <CategoryDrawer
        open={drawer.open}
        kind={activeKind}
        record={drawer.record}
        parent={drawer.parent}
        onClose={(refresh) => { setDrawer({ open: false }); if (refresh) actionRef.current?.reload(); }}
      />
    </ContentContainer>
  );
};

function stripPagination(p: any) {
  const { current, pageSize, kind, ...rest } = p ?? {};
  return rest;
}

export default CategoryManagement;
```

### 3.2 搜索/过滤体验

ProTable 搜索栏自动呈现：

- **code**（输入 `3001` 立即过滤出该子树全部 4 层节点——服务端用 `code LIKE '3001%'`）
- **name**（模糊匹配）
- **level**（下拉：大类/中类/小类/细类）
- **isEnabled**（启停）

> 进阶：将 `params.code` 字段转交服务端时映射为 `code_prefix` 过滤器（[03 §6](./03-API与Proto设计.md#6-列表查询---通用过滤器约定)），实现"输入前缀=子树过滤"的自然体验。

### 3.3 `CategoryDrawer.tsx`（新建/编辑抽屉）

用 `DrawerForm` + `formRef`（无 `useForm`，遵循 react CLAUDE.md）。两种打开方式：

| 入口 | 模式 | `kind` | `parent` | `level` | `code` |
|------|------|--------|----------|---------|--------|
| toolBar「新建大类」 | 新建（level=1） | 当前 Tab kind，只读 | 无 | 1，只读 | 2 位输入，正则校验 |
| 行内「新建子分类」 | 新建（level=parent.level+1） | 继承自父，只读 | 已知，展示为 `code · name`，只读 | parent.level+1，只读 | `parent.code + 2 位`，前缀校验 |
| 行内「编辑」 | 编辑 | 只读 | 只读 | 只读 | 只读 |
| 任何模式 | 可编辑 | — | — | — | name / nameEn / icon / description / sortOrder / isEnabled |

**前端即时校验**（与后端 [04 §6](./04-后端实现设计.md#6-关键校验汇总单元测试一对一映射) 一致，给即时反馈）：

| 字段 | 规则 |
|------|------|
| code | 纯数字、长度 = level × 2、若 parent 存在则必须 = parent.code + 2 位数字 |
| name | 必填，最长 64 字符 |

抽屉表单核心代码（伪示意）：

```tsx
<DrawerForm onFinish={async (values) => { /* create or update */ }}>
  <Form.Item name="kind" label={t('fields.kind')}>
    <Input value={kind} disabled />
  </Form.Item>
  <Form.Item label={t('fields.parent')}>
    <Input value={parent ? `${parent.code} · ${parent.name}` : t('tips.topLevel')} disabled />
  </Form.Item>
  <Form.Item name="level" label={t('fields.level')}>
    <InputNumber value={level} disabled />
  </Form.Item>
  <Form.Item
    name="code"
    label={t('fields.code')}
    rules={[
      { required: true },
      { pattern: /^\d+$/, message: t('tips.codeFormat') },
      () => ({
        validator(_, v) {
          const need = level * 2;
          if (v && v.length !== need) return Promise.reject(t('tips.codeLength', { need }));
          if (parent && v && (!v.startsWith(parent.code) || v.length !== parent.code.length + 2))
            return Promise.reject(t('tips.codePrefix', { prefix: parent.code }));
          return Promise.resolve();
        },
      }),
    ]}
  >
    <Input placeholder={parent ? `${parent.code}__` : '__'} disabled={isEditMode} />
  </Form.Item>
  {/* name / nameEn / icon / description / sortOrder / isEnabled */}
</DrawerForm>
```

## 8. 前端注意事项（遵守 react CLAUDE.md）

1. `PaginationQuery` 必须 `new`。
2. 非组件环境（路由守卫等）只用 `fetchListCategories`，不用 `useXxx`。
3. `DrawerForm` 用 `formRef`，无 `useForm`。
4. `ProTable` 的 `scroll.y` 用 `useProTableScrollY` 返回的像素值。
5. 硬编码文本全部走 i18n。
6. `meta.title` 用 `'routes:xxx'` 格式。
7. 禁止手改 `api/generated/`。
8. **默认排序按 `code` 升序**——这是 v2 设计的关键，确保层级视觉效果。

## 9. UX 流转图

```
[用户进入分类管理页]
       │
       ▼
[默认 Tab: SYSTEM] ──── 触发 List(kind=SYSTEM, order by code asc)
       │
       ▼
[ProTable 扁平展示约 250 行 SYSTEM 节点，按 code 升序自然分层]
       │
       │  用户操作
       ▼
   ┌────────────────────────────────────────────────────────┐
   │  toolBar 「新建大类」    → Drawer(level=1, parent=null)  │
   │  行操作 「新建子分类」   → Drawer(level=parent.level+1)   │
   │  行操作 「编辑」        → Drawer(record=row)             │
   │  行操作 「删除」        → Popconfirm → DELETE            │
   │  搜索栏 输入「3001」    → 服务端过滤 code LIKE '3001%'   │
   │  搜索栏 选「细类」      → 服务端过滤 level=4              │
   └────────────────────────────────────────────────────────┘
       │
       ▼
[Tabs 切到 SPACE / FACILITY，刷新表格，同上流程]
```

→ **整个交互无需 Tree 组件、无需 Splitter 分栏**，code 字符串排序 + 缩进展示就是最直观的层级表达。
