### 1.2 enum 转换器（关键，dict 无此步）

`Unit.conversion_type` 是 enum 字段，Repo 必须注册 `EnumTypeConverter`（参照 `internal_message_recipient_repo.go` 的 `statusConverter`），否则 `CopierMapper` 在 DTO↔Entity 转换时会失败：

```go
conversionTypeConverter := mapper.NewEnumTypeConverter[
    thingmodelV1.ConversionType, unit.ConversionType,
](thingmodelV1.ConversionType_name, thingmodelV1.ConversionType_value)
r.mapper.AppendConverters(conversionTypeConverter.NewConverterPair())
```

### 1.3 基准单位唯一性（partial unique index）

`OnDelete: Cascade` 已在 edge 注解表达（删分类级联删单位）。但"每分类仅一个 is_base"需额外迁移脚本：

```sql
-- 追加到迁移脚本（Ent 生成后手动补，或放 data/migrate/）
CREATE UNIQUE INDEX IF NOT EXISTS uix_thingmodel_unit_base
  ON thingmodel_units (tenant_id, category_id) WHERE is_base = true;
```

### 2.2 `unit_repo.go` 骨架

```go
package data

import (
    "context"
    "time"

    "entgo.io/ent/dialect/sql"
    "github.com/go-kratos/kratos/v2/log"
    "github.com/tx7do/kratos-bootstrap/bootstrap"

    paginationV1 "github.com/tx7do/go-crud/api/gen/go/pagination/v1"
    entCrud "github.com/tx7do/go-crud/entgo"
    "github.com/tx7do/go-utils/copierutil"
    "github.com/tx7do/go-utils/mapper"

    "go-wind-admin/app/admin/service/internal/data/ent"
    "go-wind-admin/app/admin/service/internal/data/ent/unit"
    "go-wind-admin/app/admin/service/internal/data/ent/predicate"

    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"
)

type UnitRepo struct {
    entClient *entCrud.EntClient[*ent.Client]
    log       *log.Helper

    mapper                *mapper.CopierMapper[thingmodelV1.Unit, ent.Unit]
    conversionTypeConverter *mapper.EnumTypeConverter[thingmodelV1.ConversionType, unit.ConversionType]

    repository *entCrud.Repository[
        ent.UnitQuery, ent.UnitSelect,
        ent.UnitCreate, ent.UnitCreateBulk,
        ent.UnitUpdate, ent.UnitUpdateOne,
        ent.UnitDelete,
        predicate.Unit,
        thingmodelV1.Unit, ent.Unit,
    ]
}

func NewUnitRepo(ctx *bootstrap.Context, entClient *entCrud.EntClient[*ent.Client]) *UnitRepo {
    repo := &UnitRepo{
        log:          ctx.NewLoggerHelper("unit/repo/admin-service"),
        entClient:    entClient,
        mapper:       mapper.NewCopierMapper[thingmodelV1.Unit, ent.Unit](),
        conversionTypeConverter: mapper.NewEnumTypeConverter[
            thingmodelV1.ConversionType, unit.ConversionType,
        ](thingmodelV1.ConversionType_name, thingmodelV1.ConversionType_value),
    }
    repo.init()
    return repo
}

func (r *UnitRepo) init() {
    r.repository = entCrud.NewRepository[
        ent.UnitQuery, ent.UnitSelect, ent.UnitCreate, ent.UnitCreateBulk,
        ent.UnitUpdate, ent.UnitUpdateOne, ent.UnitDelete,
        predicate.Unit, thingmodelV1.Unit, ent.Unit,
    ](r.mapper)
    r.mapper.AppendConverters(copierutil.NewTimeStringConverterPair())
    r.mapper.AppendConverters(copierutil.NewTimeTimestamppbConverterPair())
    r.mapper.AppendConverters(r.conversionTypeConverter.NewConverterPair())
}

// Count / List / IsExist / Get / Create / Update / Delete / BatchDelete
// —— 完全镜像 dict_type_repo.go 的实现（ListWithPaging、repository.Get、tx 模板等）
// Create/Update 的 SetNillable* 需覆盖：CategoryID/Code/Symbol/Name/NameEn/IsBase/
//   ConversionType/Factor/Offset/FormulaExpr/Precision/IsSiUnit/IsLegalUnit/SortOrder/IsEnabled/TenantID

// ===== 换算专用查询 =====

// GetForConvert 按 id 或 code 取单位（含换算字段）
func (r *UnitRepo) GetForConvert(ctx context.Context, req *thingmodelV1.ConvertUnitRequest, useSource bool) (*thingmodelV1.Unit, error) {
    builder := r.entClient.Client().Unit.Query()
    switch {
    case useSource && req.GetSourceUnitId() != 0:
        builder.Where(unit.IDEQ(req.GetSourceUnitId()))
    case useSource && req.GetSourceUnitCode() != "":
        builder.Where(unit.CodeEQ(req.GetSourceUnitCode()))
    case !useSource && req.GetTargetUnitId() != 0:
        builder.Where(unit.IDEQ(req.GetTargetUnitId()))
    case !useSource && req.GetTargetUnitCode() != "":
        builder.Where(unit.CodeEQ(req.GetTargetUnitCode()))
    default:
        return nil, thingmodelV1.ErrorBadRequest("source/target unit required")
    }
    // 直接取 entity（需 factor/offset/category_id），不走 view_mask
    return r.repository.Get(ctx, builder, nil /* viewMask=nil */)
}

// ListByCategory 某分类下全部单位（下拉选择器）
func (r *UnitRepo) ListByCategory(ctx context.Context, req *thingmodelV1.ListUnitByCategoryRequest) (*thingmodelV1.ListUnitResponse, error) {
    builder := r.entClient.Client().Unit.Query()
    if req.GetCategoryId() != 0 {
        builder.Where(unit.CategoryIDEQ(req.GetCategoryId()))
    } else if req.GetCategoryCode() != "" {
        // 需先查 category id（略，或 join）
    }
    if req.GetOnlyEnabled() {
        builder.Where(unit.IsEnabledEQ(true))
    }
    builder.Order(ent.Asc(unit.FieldSortOrder), ent.Asc(unit.FieldID))
    // ... 走 repository 或直接 All + mapper 转 DTO
}
```

### 3.3 `unit_service.go`（核心）

```go
package service

import (
    "context"
    "fmt"
    "math"

    "github.com/go-kratos/kratos/v2/log"
    paginationV1 "github.com/tx7do/go-crud/api/gen/go/pagination/v1"
    "github.com/tx7do/go-utils/trans"
    "github.com/tx7do/kratos-bootstrap/bootstrap"
    "google.golang.org/protobuf/types/known/emptypb"

    "go-wind-admin/app/admin/service/internal/data"
    adminV1 "go-wind-admin/api/gen/go/admin/service/v1"
    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"
    "go-wind-admin/pkg/middleware/auth"
)

type UnitService struct {
    adminV1.UnitServiceHTTPServer

    log    *log.Helper
    repo   *data.UnitRepo

    // 依赖分类 repo（用于校验 category 存在性、查 base unit）
    catRepo *data.UnitCategoryRepo
}

func NewUnitService(ctx *bootstrap.Context, repo *data.UnitRepo, catRepo *data.UnitCategoryRepo) *UnitService {
    return &UnitService{
        log: ctx.NewLoggerHelper("unit/service/admin-service"),
        repo: repo, catRepo: catRepo,
    }
}

// List / Get / ListByCategory —— 透传 repo（略，镜像 dict_type_service）

// Create 创建单位（含换算一致性校验）
func (s *UnitService) Create(ctx context.Context, req *thingmodelV1.CreateUnitRequest) (*emptypb.Empty, error) {
    if req.Data == nil {
        return nil, thingmodelV1.ErrorBadRequest("invalid parameter")
    }
    if err := s.validateUnit(req.Data); err != nil {
        return nil, err
    }
    operator, err := auth.FromContext(ctx)
    if err != nil {
        return nil, err
    }
    req.Data.CreatedBy = trans.Ptr(operator.UserId)
    if err := s.repo.Create(ctx, req); err != nil {
        return nil, err
    }
    return &emptypb.Empty{}, nil
}

// Update 更新单位（含校验 + 基准单位切换事务）
func (s *UnitService) Update(ctx context.Context, req *thingmodelV1.UpdateUnitRequest) (*emptypb.Empty, error) {
    if req.Data == nil {
        return nil, thingmodelV1.ErrorBadRequest("invalid parameter")
    }
    if err := s.validateUnit(req.Data); err != nil {
        return nil, err
    }
    operator, err := auth.FromContext(ctx)
    if err != nil {
        return nil, err
    }
    req.Data.Id = trans.Ptr(req.GetId())
    req.Data.UpdatedBy = trans.Ptr(operator.UserId)
    if req.UpdateMask != nil {
        req.UpdateMask.Paths = append(req.UpdateMask.Paths, "updated_by")
    }
    // 若把 is_base 从 false 改 true，需事务内先将同分类旧 base 置 false（见 3.4）
    if err := s.repo.Update(ctx, req); err != nil {
        return nil, err
    }
    return &emptypb.Empty{}, nil
}

// Delete 删除（被引用时拒绝物理删除）
func (s *UnitService) Delete(ctx context.Context, req *thingmodelV1.DeleteUnitRequest) (*emptypb.Empty, error) {
    // 本期 reference_count 恒 0，预留校验位：
    // for _, id := range req.GetIds() { if u, _ := s.repo.Get(...); u.ReferenceCount > 0 { return UNIT_IN_USE_CANNOT_DELETE } }
    if err := s.repo.BatchDelete(ctx, req.GetIds()); err != nil {
        return nil, err
    }
    return &emptypb.Empty{}, nil
}

// ===== 换算核心 =====
func (s *UnitService) Convert(ctx context.Context, req *thingmodelV1.ConvertUnitRequest) (*thingmodelV1.ConvertUnitResponse, error) {
    src, err := s.repo.GetForConvert(ctx, req, true)
    if err != nil || src == nil {
        return s.convFail(thingmodelV1.ConvertUnitResponse_CONVERT_NOT_FOUND, "source unit not found"), nil
    }
    dst, err := s.repo.GetForConvert(ctx, req, false)
    if err != nil || dst == nil {
        return s.convFail(thingmodelV1.ConvertUnitResponse_CONVERT_NOT_FOUND, "target unit not found"), nil
    }
    if src.CategoryId == nil || dst.CategoryId == nil || src.GetCategoryId() != dst.GetCategoryId() {
        return s.convFail(thingmodelV1.ConvertUnitResponse_CONVERT_DIFFERENT_CATEGORY,
            fmt.Sprintf("%s 与 %s 不属于同一物理量分类", src.GetCode(), dst.GetCode())), nil
    }
    if !isConvertible(src.ConversionType) || !isConvertible(dst.ConversionType) {
        return s.convFail(thingmodelV1.ConvertUnitResponse_CONVERT_NOT_CONVERTIBLE,
            notConvertibleReason(src, dst)), nil
    }
    srcFactor, srcOffset := coef(src)
    dstFactor, dstOffset := coef(dst)
    if srcFactor == 0 || dstFactor == 0 {
        return nil, thingmodelV1.ErrorUnitInvalidFactor("factor must not be zero")
    }

    // base = value·srcFactor + srcOffset
    base := req.GetValue()*srcFactor + srcOffset
    // result = (base - dstOffset) / dstFactor
    result := (base - dstOffset) / dstFactor

    if math.IsNaN(result) || math.IsInf(result, 0) {
        return nil, thingmodelV1.ErrorUnitOverflow("convert result overflow")
    }

    precision := int(req.GetPrecision())
    if precision == 0 && req.Precision == nil {
        precision = int(dst.GetPrecision()) // 用目标单位建议精度
    }
    result = roundTo(result, precision)

    formula := fmt.Sprintf("%v %s → %v %s → %v %s",
        req.GetValue(), src.GetSymbol(),
        roundTo(base, precision), src.GetSymbol(), // 经基准中转（基准符号取同分类 base）
        result, dst.GetSymbol())

    return &thingmodelV1.ConvertUnitResponse{
        Result: result, Formula: formula,
        Status: thingmodelV1.ConvertUnitResponse_CONVERT_OK,
        BaseValue: trans.Ptr(roundTo(base, precision)),
    }, nil
}

// ===== 校验（见 03 章 4 节规则）=====
func (s *UnitService) validateUnit(u *thingmodelV1.Unit) error {
    if u.GetCode() == "" || u.GetSymbol() == "" || u.GetName() == "" {
        return thingmodelV1.ErrorBadRequest("code/symbol/name required")
    }
    if u.GetCategoryId() == 0 {
        return thingmodelV1.ErrorUnitCategoryNotFound("category_id required")
    }
    f, o := u.GetFactor(), u.GetOffset()
    if f == 0 {
        f = 1 // 缺省视为 1
    }
    switch u.GetConversionType() {
    case thingmodelV1.ConversionType_LINEAR:
        if o != 0 {
            return thingmodelV1.ErrorUnitLinearOffsetMustBeZero("linear unit offset must be 0")
        }
    }
    if u.GetIsBase() {
        if f != 1 || o != 0 {
            return thingmodelV1.ErrorUnitBaseFactorInvalid("base unit factor=1 & offset=0 required")
        }
    }
    if f == 0 {
        return thingmodelV1.ErrorUnitFactorZero("factor must not be zero")
    }
    return nil
}

// ===== 辅助 =====
func isConvertible(t thingmodelV1.ConversionType) bool {
    return t == thingmodelV1.ConversionType_LINEAR || t == thingmodelV1.ConversionType_AFFINE
}
func coef(u *thingmodelV1.Unit) (factor, offset float64) {
    if u.Factor == nil || u.GetFactor() == 0 {
        factor = 1
    } else {
        factor = u.GetFactor()
    }
    offset = u.GetOffset()
    return
}
func notConvertibleReason(src, dst *thingmodelV1.Unit) string {
    // 根据 conversion_type 给出对数/条件/无量纲的具体提示（略）
    return "unit not linearly convertible"
}
func roundTo(v float64, p int) float64 {
    if p < 0 {
        p = 0
    }
    pow := math.Pow(10, float64(p))
    return math.Round(v*pow) / pow
}
func (s *UnitService) convFail(st thingmodelV1.ConvertUnitResponse_ConvertUnitStatus, msg string) *thingmodelV1.ConvertUnitResponse {
    return &thingmodelV1.ConvertUnitResponse{Status: st, Message: msg}
}
```

### 3.4 基准单位切换事务

当 Update 把 `is_base` 从 false 改为 true 时，需在同一事务内把同分类原基准置 false。建议在 `unit_repo.go.Update` 内，当检测到 `is_base` 进入 update_mask 且新值为 true 时：

```go
// 伪代码（嵌在 Update 的 tx 内，builder 构建前）
if maskContains(req.UpdateMask, "is_base") && req.Data.GetIsBase() {
    // 先把同 (tenant_id, category_id) 下现有 base 置 false
    tx.Unit.Update().
        Where(unit.CategoryIDEQ(req.Data.GetCategoryId()),
              unit.TenantIDEQ(req.Data.GetTenantId()),
              unit.IsBaseEQ(true)).
        SetIsBase(false).Exec(ctx)
}
```
