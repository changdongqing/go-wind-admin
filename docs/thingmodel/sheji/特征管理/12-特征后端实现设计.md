### 1.2 enum 转换器（关键，与 unit 同模式）

Feature 有 **5 个 enum 字段**（feature_type / data_type / access_mode / event_level / call_mode），Repo 必须为每个注册 `EnumTypeConverter`（参照 `unit_repo.go` 的 `convTypeConverter`）：

```go

r.dataTypeConverter = mapper.NewEnumTypeConverter[
    thingmodelV1.DataType, feature.DataType,
](thingmodelV1.DataType_name, thingmodelV1.DataType_value)

r.accessModeConverter = mapper.NewEnumTypeConverter[
    thingmodelV1.AccessMode, feature.AccessMode,
](thingmodelV1.AccessMode_name, thingmodelV1.AccessMode_value)

r.eventLevelConverter = mapper.NewEnumTypeConverter[
    thingmodelV1.EventLevel, feature.EventLevel,
](thingmodelV1.EventLevel_name, thingmodelV1.EventLevel_value)

r.callModeConverter = mapper.NewEnumTypeConverter[
    thingmodelV1.CallMode, feature.CallMode,
](thingmodelV1.CallMode_name, thingmodelV1.CallMode_value)

r.mapper.AppendConverters(
    r.featureTypeConverter.NewConverterPair(),
    r.dataTypeConverter.NewConverterPair(),
    r.accessModeConverter.NewConverterPair(),
    r.eventLevelConverter.NewConverterPair(),
    r.callModeConverter.NewConverterPair(),
)
```

### 1.3 spec 字段映射（强类型 FeatureSpec ↔ JSON）

proto `Feature.spec` 与 ent 字段都用强类型 `*thingmodelV1.FeatureSpec`（ent：`field.JSON("spec", &thingmodelV1.FeatureSpec{})`）。Ent 的 `field.JSON` 对 proto message 类型自动 marshal/unmarshal（项目 `menu.meta` 用 `&permissionV1.MenuMeta{}` 即此模式），CopierMapper 对同类型 DTO↔Entity 直接透传，**无需额外转换器**：

```go
// FeatureSpec 在 DTO(thingmodelV1.Feature.spec) 与 Entity(ent.Feature.spec) 两侧类型一致，
// CopierMapper 直接赋值透传，无需注册 Struct/Map 转换器。
// 若 go-crud 对 proto message 的 JSON 字段需要显式 converter，参照 menu 模块处理方式即可。
```

```go
package data

import (
    "context"

    paginationV1 "github.com/tx7do/go-crud/api/gen/go/pagination/v1"
    entCrud "github.com/tx7do/go-crud/entgo"
    "github.com/tx7do/go-utils/copierutil"
    "github.com/tx7do/go-utils/mapper"

    "go-wind-admin/app/admin/service/internal/data/ent"
    "go-wind-admin/app/admin/service/internal/data/ent/feature"
    "go-wind-admin/app/admin/service/internal/data/ent/predicate"

    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"
)

// FeatureRepo 特征仓库 / Feature repository
type FeatureRepo struct {
    entClient *entCrud.EntClient[*ent.Client]
    log       *log.Helper

    mapper *mapper.CopierMapper[thingmodelV1.Feature, ent.Feature]

    // 5 个 enum 转换器
    featureTypeConverter *mapper.EnumTypeConverter[thingmodelV1.FeatureType, feature.FeatureType]
    dataTypeConverter    *mapper.EnumTypeConverter[thingmodelV1.DataType, feature.DataType]
    accessModeConverter  *mapper.EnumTypeConverter[thingmodelV1.AccessMode, feature.AccessMode]
    eventLevelConverter  *mapper.EnumTypeConverter[thingmodelV1.EventLevel, feature.EventLevel]
    callModeConverter    *mapper.EnumTypeConverter[thingmodelV1.CallMode, feature.CallMode]

    repository *entCrud.Repository[
        ent.FeatureQuery, ent.FeatureSelect,
        ent.FeatureCreate, ent.FeatureCreateBulk,
        ent.FeatureUpdate, ent.FeatureUpdateOne,
        ent.FeatureDelete,
        predicate.Feature,
        thingmodelV1.Feature, ent.Feature,
    ]
}

func NewFeatureRepo(ctx *bootstrap.Context, entClient *entCrud.EntClient[*ent.Client]) *FeatureRepo {
    repo := &FeatureRepo{
        log:       ctx.NewLoggerHelper("feature/repo/admin-service"),
        entClient: entClient,
        mapper:    mapper.NewCopierMapper[thingmodelV1.Feature, ent.Feature](),
        // 初始化 5 个 enum 转换器（见 §1.2）
    }
    repo.init()
    return repo
}

func (r *FeatureRepo) init() {
    r.repository = entCrud.NewRepository[
        ent.FeatureQuery, ent.FeatureSelect, ent.FeatureCreate, ent.FeatureCreateBulk,
        ent.FeatureUpdate, ent.FeatureUpdateOne, ent.FeatureDelete,
        predicate.Feature, thingmodelV1.Feature, ent.Feature,
    ](r.mapper)

    r.mapper.AppendConverters(copierutil.NewTimeStringConverterPair())
    r.mapper.AppendConverters(copierutil.NewTimeTimestamppbConverterPair())
    // 注册 5 个 enum 转换器（见 §1.2）
    // 注册 Struct↔map 转换器（见 §1.3）
}

// Count / List / IsExist / Get / Create / Update / Delete / BatchDelete
// —— 完全镜像 unit_repo.go 的实现（ListWithPaging、repository.Get、tx 模板等）
// Create/Update 的 SetNillable* 需覆盖：
//   FeatureType/Code/Identifier/Name/NameEn/Description/ApplicableScope/
//   DataType/AccessMode/EventLevel/CallMode/RelationType/Spec/SortOrder/IsEnabled/TenantID

// ===== 按类型查询 =====

// ListByType 某类型下全部特征（左侧树联动）
func (r *FeatureRepo) ListByType(ctx context.Context, req *thingmodelV1.ListFeatureByTypeRequest) (*thingmodelV1.ListFeatureResponse, error) {
    builder := r.entClient.Client().Feature.Query()
    if req.GetFeatureType() != thingmodelV1.FeatureType_FEATURE_TYPE_UNSPECIFIED {
        builder.Where(feature.FeatureTypeEQ(feature.FeatureType(req.GetFeatureType().String())))
    }
    if req.GetOnlyEnabled() {
        builder.Where(feature.IsEnabledEQ(true))
    }
    if req.GetApplicableScope() != "" {
        builder.Where(feature.ApplicableScopeEQ(req.GetApplicableScope()))
    }
    builder.Order(ent.Asc(feature.FieldSortOrder), ent.Asc(feature.FieldID))
    // ... 走 repository 或直接 All + mapper 转 DTO
}

// ReferencedByRelation 判断指定 feature 是否被某 relation 引用（source/target）
// 用于删除前的完整性校验（约束 F11）
func (r *FeatureRepo) ReferencedByRelation(ctx context.Context, id uint32) (bool, error) {
    // 查 feature_type=RELATION 且 spec JSON 内 source.id 或 target.id = id
    // 用 SQL JSON 查询：spec->'source'->>'id' = :id OR spec->'target'->>'id' = :id
    count, err := r.entClient.Client().Feature.Query().
        Where(feature.FeatureTypeEQ(feature.FeatureTypeRELATION)).
        Modify(func(s *sql.Selector) {
            s.Where(sql.ExprP("(spec->'source'->>'id' = ? OR spec->'target'->>'id' = ?)", id, id))
        }).
        Count(ctx)
    return count > 0, err
}
```

> `ReferencedByRelation` 用 PostgreSQL 的 JSON 查询（`spec->'source'->>'id'`）。若 DB 是 MySQL，语法调整为 `JSON_EXTRACT(spec, '$.source.id')`。参照项目 `data.yaml` 确认 DB 类型。

### 3.2 `feature_service.go`（核心）

```go
package service

import (
    "context"

    "github.com/go-kratos/kratos/v2/log"
    paginationV1 "github.com/tx7do/go-crud/api/gen/go/pagination/v1"
    "github.com/tx7do/go-utils/trans"
    "github.com/tx7do/kratos-bootstrap/bootstrap"
    "google.golang.org/protobuf/types/known/emptypb"

    "go-wind-admin/app/admin/service/internal/data"

    adminV1 "go-wind-admin/api/gen/go/admin/service/v1"
    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"

    "go-wind-admin/pkg/middleware/auth"
)

// FeatureService 特征服务 / Feature service
type FeatureService struct {
    adminV1.FeatureServiceHTTPServer

    log    *log.Helper
    repo   *data.FeatureRepo
    unitRepo *data.UnitRepo   // 维护 unit.reference_count（打通单位管理挂钩点）
}

func NewFeatureService(
    ctx *bootstrap.Context,
    repo *data.FeatureRepo,
    unitRepo *data.UnitRepo,
) *FeatureService {
    return &FeatureService{
        log:      ctx.NewLoggerHelper("feature/service/admin-service"),
        repo:     repo,
        unitRepo: unitRepo,
    }
}

// List / Get / ListByType —— 透传 repo（略，镜像 unit_service.go）

// ValidateSpec 校验 spec（不落库）/ Validate spec without persisting
func (s *FeatureService) ValidateSpec(ctx context.Context, req *thingmodelV1.ValidateFeatureSpecRequest) (*thingmodelV1.ValidateFeatureSpecResponse, error) {
    errs := validateFeatureSpec(req.GetSpec())
    return &thingmodelV1.ValidateFeatureSpecResponse{
        Valid:  len(errs) == 0,
        Errors: errs,
    }, nil
}

// Create 创建特征 / Create feature
func (s *FeatureService) Create(ctx context.Context, req *thingmodelV1.CreateFeatureRequest) (*emptypb.Empty, error) {
    if req.Data == nil {
        return nil, thingmodelV1.ErrorBadRequest("invalid parameter")
    }
    // 1. 校验 spec（按 oneof 分支分派）
    if errs := validateFeatureSpec(req.Data.GetSpec()); len(errs) > 0 {
        return nil, thingmodelV1.ErrorFeatureSpecInvalid("%v", errs)
    }
    // 2. 同步特化列（spec → 抽取列，保证一致性，约束 F4/F17）
    syncSpecializedColumns(req.Data)

    operator, err := auth.FromContext(ctx)
    if err != nil {
        return nil, err
    }
    req.Data.CreatedBy = trans.Ptr(operator.UserId)

    if err := s.repo.Create(ctx, req); err != nil {
        return nil, err
    }

    // 3. 若是 property 且引用了 unit，维护 unit.reference_count +1（约束 F6/F10）
    s.adjustUnitReference(ctx, req.Data, +1)

    return &emptypb.Empty{}, nil
}

// Update 更新特征 / Update feature
func (s *FeatureService) Update(ctx context.Context, req *thingmodelV1.UpdateFeatureRequest) (*emptypb.Empty, error) {
    if req.Data == nil {
        return nil, thingmodelV1.ErrorBadRequest("invalid parameter")
    }
    if errs := validateFeatureSpec(req.Data.GetSpec()); len(errs) > 0 {
        return nil, thingmodelV1.ErrorFeatureSpecInvalid("%v", errs)
    }
    syncSpecializedColumns(req.Data)

    operator, err := auth.FromContext(ctx)
    if err != nil {
        return nil, err
    }
    req.Data.Id = trans.Ptr(req.GetId())
    req.Data.UpdatedBy = trans.Ptr(operator.UserId)
    if req.UpdateMask != nil {
        req.UpdateMask.Paths = append(req.UpdateMask.Paths, "updated_by")
    }

    // 若 property 的 unit 引用变更，调整旧/新 unit 的 reference_count
    s.handleUnitReferenceChange(ctx, req)

    if err := s.repo.Update(ctx, req); err != nil {
        return nil, err
    }
    return &emptypb.Empty{}, nil
}

// Delete 批量删除 / Batch delete
func (s *FeatureService) Delete(ctx context.Context, req *thingmodelV1.DeleteFeatureRequest) (*emptypb.Empty, error) {
    ids := req.GetIds()
    if len(ids) == 0 {
        return nil, thingmodelV1.ErrorBadRequest("ids is required")
    }

    // 关系完整性校验（约束 F11）：被 relation 引用的 feature 拒绝删除
    for _, id := range ids {
        referenced, err := s.repo.ReferencedByRelation(ctx, id)
        if err != nil {
            return nil, err
        }
        if referenced {
            return nil, thingmodelV1.ErrorFeatureInUseCannotDelete(
                "feature %d is referenced by relation(s), remove relations first", id)
        }
    }

    // 维护 unit.reference_count -1（被删 property 引用的 unit）
    for _, id := range ids {
        feat, _ := s.repo.Get(ctx, &thingmodelV1.GetFeatureRequest{
            QueryBy: &thingmodelV1.GetFeatureRequest_Id{Id: id},
        })
        if feat != nil {
            s.adjustUnitReference(ctx, feat, -1)
        }
    }

    if err := s.repo.BatchDelete(ctx, ids); err != nil {
        return nil, err
    }
    return &emptypb.Empty{}, nil
}

// ===== 特化列同步（spec → 抽取列）=====
// syncSpecializedColumns 将 spec oneof 分支内的字段同步到特化列，保证一致性（约束 F4/F17）
func syncSpecializedColumns(f *thingmodelV1.Feature) {
    if f.Spec == nil {
        return
    }
    switch spec := f.Spec.Spec.(type) {        // oneof 取分支
    case *thingmodelV1.FeatureSpec_Property:
        f.DataType = spec.Property.GetDataType()
        f.AccessMode = spec.Property.GetAccessMode()
    case *thingmodelV1.FeatureSpec_Event:
        f.EventLevel = spec.Event.GetLevel()
    case *thingmodelV1.FeatureSpec_Service:
        f.CallMode = spec.Service.GetCallMode()
    case *thingmodelV1.FeatureSpec_Relation:
        f.RelationType = trans.Ptr(spec.Relation.GetRelationType())
    }
}

// ===== 单位引用计数维护（打通单位管理挂钩点）=====
func (s *FeatureService) adjustUnitReference(ctx context.Context, f *thingmodelV1.Feature, delta int32) {
    if f.GetFeatureType() != thingmodelV1.FeatureType_PROPERTY || f.Spec == nil {
        return
    }
    prop, ok := f.Spec.Spec.(*thingmodelV1.FeatureSpec_Property)
    if !ok || prop.Property.GetUnit() == nil {
        return
    }
    unitID := prop.Property.GetUnit().GetUnitId()
    if unitID == 0 {
        return
    }
    // 调用 unit_repo 维护 reference_count（需在 unit_repo 新增 IncReferenceCount 方法）
    if err := s.unitRepo.IncReferenceCount(ctx, unitID, delta); err != nil {
        s.log.Errorf("adjust unit %d reference_count failed: %v", unitID, err)
        // 不阻断主流程（特征已创建），仅记日志；由对账任务兜底
    }
}
```

> **`IncReferenceCount` 方法**：需在 `unit_repo.go` 新增一个 `IncReferenceCount(ctx, id, delta)` 方法（`UPDATE thingmodel_units SET reference_count = reference_count + ? WHERE id = ?`），兑现单位管理预留的挂钩点。

### 3.3 `feature_validator.go`（四类 spec 校验器）

```go
package service

import (
    thingmodelV1 "go-wind-admin/api/gen/go/thingmodel/service/v1"
)

// validateFeatureSpec 按 FeatureSpec oneof 分派校验（规则见 10 章 §6）
func validateFeatureSpec(spec *thingmodelV1.FeatureSpec) []string {
    if spec == nil || spec.Spec == nil {
        return []string{"spec is required"}
    }
    switch s := spec.Spec.(type) {
    case *thingmodelV1.FeatureSpec_Property:
        return validatePropertySpec(s.Property)
    case *thingmodelV1.FeatureSpec_Event:
        return validateEventSpec(s.Event)
    case *thingmodelV1.FeatureSpec_Service:
        return validateServiceSpec(s.Service)
    case *thingmodelV1.FeatureSpec_Relation:
        return validateRelationSpec(s.Relation)
    default:
        return []string{"unknown spec type"}
    }
}

// validatePropertySpec 属性 spec 校验（规则 V1~V9）
func validatePropertySpec(p *thingmodelV1.PropertySpec) []string {
    var errs []string
    dt := p.GetDataType()
    if dt == thingmodelV1.DataType_DATA_TYPE_UNSPECIFIED {
        errs = append(errs, "property.dataType required")
    }
    if p.GetAccessMode() == thingmodelV1.AccessMode_ACCESS_MODE_UNSPECIFIED {
        errs = append(errs, "property.accessMode required")
    }
    // V3: dataType=ENUM 时 enumItems 非空
    if dt == thingmodelV1.DataType_ENUM && len(p.GetEnumItems()) == 0 {
        errs = append(errs, "property.enumItems required for ENUM type")
    }
    // V4: dataType=STRUCT 时 structFields 非空
    if dt == thingmodelV1.DataType_STRUCT && len(p.GetStructFields()) == 0 {
        errs = append(errs, "property.structFields required for STRUCT type")
    }
    // V5: dataType=ARRAY 时 arraySpec.element 非空
    if dt == thingmodelV1.DataType_ARRAY && (p.GetArraySpec() == nil || p.GetArraySpec().GetElement() == nil) {
        errs = append(errs, "property.arraySpec.element required for ARRAY type")
    }
    // V7: constraints.min ≤ max
    if c := p.GetConstraints(); c != nil && c.GetMin() > c.GetMax() {
        errs = append(errs, "property.constraints.min must be <= max")
    }
    // V9: category=rated 时 isRated=true
    if p.GetCategory() == "rated" && !p.GetIsRated() {
        errs = append(errs, "property.isRated must be true when category=rated")
    }
    return errs
}

// validateEventSpec 事件 spec 校验（V10）
func validateEventSpec(e *thingmodelV1.EventSpec) []string {
    var errs []string
    if e.GetLevel() == thingmodelV1.EventLevel_EVENT_LEVEL_UNSPECIFIED {
        errs = append(errs, "event.level required")
    }
    return errs
}

// validateServiceSpec 服务 spec 校验（V11）
func validateServiceSpec(s *thingmodelV1.ServiceSpec) []string {
    var errs []string
    if s.GetCallMode() == thingmodelV1.CallMode_CALL_MODE_UNSPECIFIED {
        errs = append(errs, "service.callMode required")
    }
    return errs
}

// validateRelationSpec 关系 spec 校验（V13~V15）
func validateRelationSpec(r *thingmodelV1.RelationSpec) []string {
    var errs []string
    if r.GetRelationType() == "" {
        errs = append(errs, "relation.relationType required")
    }
    if r.GetSource() == nil || r.GetTarget() == nil {
        errs = append(errs, "relation.source and target required")
    }
    // V15: source.kind=feature 时目标须存在（此处仅结构校验，存在性校验在 Service 内查 DB）
    return errs
}
```

> **关系目标存在性校验（V15）**：需在 Service 层（非纯函数）内额外查 DB——若 `source.kind=feature`，校验 `source.id` 对应的 feature 存在。这是 `validateFeatureSpec` 纯函数无法完成的（无 DB 访问），故拆到 Service 内。

## 5. 单位引用计数维护（兑现单位管理挂钩点）

本模块是单位管理文档约定的 `reference_count` 维护方。需在 `unit_repo.go` 新增方法：

```go
// IncReferenceCount 调整单位引用计数（delta 可正可负）
// Adjust unit reference_count by delta (positive on property create, negative on delete)
func (r *UnitRepo) IncReferenceCount(ctx context.Context, id uint32, delta int32) error {
    return r.entClient.Client().Unit.UpdateOneID(id).
        AddReferenceCount(delta).   // ent 生成的 Add 方法
        Exec(ctx)
}
```

调用时机（在 `feature_service.go`）：
- **Create property**（含 unit 引用）：`adjustUnitReference(feat, +1)`
- **Delete property**：`adjustUnitReference(feat, -1)`
- **Update property 且 unit 变更**：旧 unit -1，新 unit +1（`handleUnitReferenceChange`）

> 兑现后，单位管理的 **C6 约束**（`reference_count > 0` 禁止物理删除）自动生效——删除被 property 引用的 unit 时，`unit_service.go` 已有的 `ReferencedIDs` 校验会拒绝。
