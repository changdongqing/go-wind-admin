## 2. Hooks 层（`api/hooks/unit.ts`）

镜像 `dict.ts`，新增换算 mutation。类型从 `@/api/generated/admin/service/v1` 导入（`make ts` 后生成 `unitservicev1_*` / `unitcategoryservicev1_*`）。

```typescript
import { useMutation, useQuery, type UseMutationOptions, type UseQueryOptions } from '@tanstack/react-query';
import {
  type unitcategoryservicev1_UnitCategory,
  type unitcategoryservicev1_ListUnitCategoryResponse,
  type unitcategoryservicev1_GetUnitCategoryRequest,
  type unitcategoryservicev1_CreateUnitCategoryRequest,
  type unitcategoryservicev1_DeleteUnitCategoryRequest,
  type unitservicev1_Unit,
  type unitservicev1_ListUnitResponse,
  type unitservicev1_GetUnitRequest,
  type unitservicev1_CreateUnitRequest,
  type unitservicev1_DeleteUnitRequest,
  type unitservicev1_ListUnitByCategoryRequest,
  type unitservicev1_ConvertUnitRequest,
  type unitservicev1_ConvertUnitResponse,
} from '@/api/generated/admin/service/v1';
import { makeUpdateMask, type PaginationQuery } from '@/core/transport/rest';
import { queryClient } from '@/core';
import { apiClient } from '@/api/client';

// ========== 物理量分类 ==========
export function useListUnitCategories(q: PaginationQuery, o?: UseQueryOptions<unitcategoryservicev1_ListUnitCategoryResponse, Error>) {
  return useQuery({ queryKey: ['listUnitCategories', q], queryFn: () => apiClient.unitCategoryService.List(q.toRawParams()), ...o });
}
export async function fetchListUnitCategories(params: PaginationQuery) {
  return queryClient.fetchQuery({ queryKey: ['listUnitCategories', params], queryFn: () => apiClient.unitCategoryService.List(params.toRawParams()), retry: 0 });
}
export function useCreateUnitCategory(o?: UseMutationOptions<{}, Error, unitcategoryservicev1_CreateUnitCategoryRequest>) {
  return useMutation({ mutationFn: (data) => apiClient.unitCategoryService.Create(data), ...o });
}
export function useUpdateUnitCategory(o?: UseMutationOptions<{}, Error, { id: number; values: Record<string, any> }>) {
  return useMutation({
    mutationFn: ({ id, values }) => apiClient.unitCategoryService.Update({ id, data: { ...values }, updateMask: makeUpdateMask(Object.keys(values ?? {})) }),
    ...o,
  });
}
export function useDeleteUnitCategory(o?: UseMutationOptions<{}, Error, unitcategoryservicev1_DeleteUnitCategoryRequest>) {
  return useMutation({ mutationFn: (data) => apiClient.unitCategoryService.Delete(data), ...o });
}

// ========== 单位 ==========
export function useListUnits(q: PaginationQuery, o?: UseQueryOptions<unitservicev1_ListUnitResponse, Error>) {
  return useQuery({ queryKey: ['listUnits', q], queryFn: () => apiClient.unitService.List(q.toRawParams()), ...o });
}
export async function fetchListUnits(params: PaginationQuery) {
  return queryClient.fetchQuery({ queryKey: ['listUnits', params], queryFn: () => apiClient.unitService.List(params.toRawParams()), retry: 0 });
}
// 按分类查询（属性选单位下拉框用，可不分页）
export function useListUnitsByCategory(req: unitservicev1_ListUnitByCategoryRequest, o?: UseQueryOptions<unitservicev1_ListUnitResponse, Error>) {
  return useQuery({ queryKey: ['listUnitsByCategory', req], queryFn: () => apiClient.unitService.ListByCategory(req), ...o });
}
export function useCreateUnit(o?: UseMutationOptions<{}, Error, unitservicev1_CreateUnitRequest>) {
  return useMutation({ mutationFn: (data) => apiClient.unitService.Create(data), ...o });
}
export function useUpdateUnit(o?: UseMutationOptions<{}, Error, { id: number; values: Record<string, any> }>) {
  return useMutation({
    mutationFn: ({ id, values }) => apiClient.unitService.Update({ id, data: { ...values } as any, updateMask: makeUpdateMask(Object.keys(values ?? {})) }),
    ...o,
  });
}
export function useDeleteUnit(o?: UseMutationOptions<{}, Error, unitservicev1_DeleteUnitRequest>) {
  return useMutation({ mutationFn: (data) => apiClient.unitService.Delete(data), ...o });
}

// ========== 单位换算 ==========
export function useConvertUnit(o?: UseMutationOptions<unitservicev1_ConvertUnitResponse, Error, unitservicev1_ConvertUnitRequest>) {
  return useMutation({ mutationFn: (data) => apiClient.unitService.Convert(data), ...o });
}
```

> **注意**：`apiClient.unitCategoryService` / `apiClient.unitService` 两个 getter 需在 `src/api/client.ts` 中按现有懒加载模式新增（生成代码已提供 client 类）。

## 3. 页面布局（`index.tsx`）

镜像 `dict/index.tsx` 的 `Splitter` 左右分栏：

```tsx
import { Splitter } from 'antd';
import { useState } from 'react';
import ContentContainer from '@/layouts/components/PageContainer/ContentContainer';
import UnitCategoryList from './UnitCategoryList';
import UnitList from './UnitList';

const UnitManagement = () => {
  const [currentCategoryId, setCurrentCategoryId] = useState<number | null>(null);
  return (
    <ContentContainer heightMode="fixed" padding="16px" bottomMargin={0}>
      <Splitter style={{ height: '100%', flex: 1, minHeight: 0 }}>
        <Splitter.Panel collapsible defaultSize="38%" min="25%" max="50%">
          <UnitCategoryList currentCategoryId={currentCategoryId} onCategorySelect={setCurrentCategoryId} />
        </Splitter.Panel>
        <Splitter.Panel>
          <UnitList categoryId={currentCategoryId} />
        </Splitter.Panel>
      </Splitter>
    </ContentContainer>
  );
};
export default UnitManagement;
```

### 3.1 `UnitCategoryList.tsx`（左）

镜像 `DictTypeList.tsx`：ProTable 列为 `name` / `code` / `baseUnitSymbol` / `isEnabled`(Tag) / `action`；`onRow` 点击选中分类并高亮（`outline`）；`toolBarRender` 新建按钮；删除用 `Popconfirm` + `useDeleteUnitCategory`；`request` 走 `fetchListUnitCategories(new PaginationQuery({...}))`。

### 3.2 `UnitList.tsx`（右）

镜像 `DictEntryList.tsx`：`categoryId` 为空时 `<Empty>`；`useEffect([categoryId])` 触发 `actionRef.reload()`；列含**单位特有字段**：

| 列 | dataIndex | 说明 |
|----|-----------|------|
| 符号 | symbol | 如 `℃`（fixed left） |
| 名称 | name | 中文名 |
| 编码 | code | 如 `celsius` |
| 基准 | isBase | Tag「基准」(is_base=true 高亮) |
| 换算类型 | conversionType | Tag（颜色见 constants） |
| 系数/偏移 | factor/offset | `x·{factor}+{offset}` 文本（hideInSearch） |
| 状态 | isEnabled | Tag |
| 操作 | — | 编辑 / 删除 / **换算**（打开换算器预填该单位） |

`request` 中 `formValues` 携带 `category_id: categoryId`。

### 3.3 `UnitDrawer.tsx`（单位新建/编辑）

用 `DrawerForm` + `formRef`（无 `useForm`，遵循 react CLAUDE.md 注意事项 6）。表单字段：

- 基础：categoryId（已锁定为当前分类，只读展示）、code、symbol、name、nameEn
- 换算区（分组）：
  - `isBase`（Switch，开启时自动锁定 factor=1/offset=0 并禁用输入）
  - `conversionType`（Select：线性/仿射/对数/条件/不可换算）
  - `factor`、`offset`（当 conversionType=LINEAR 时 offset 禁用并置 0；当 isBase 时两者禁用）
  - `formulaExpr`（展示用公式说明，TextArea）
- 展示区：precision、isSiUnit、isLegalUnit、sortOrder、isEnabled

**前端即时校验**（与后端 [05 校验](./05-后端实现设计.md) 一致，给即时反馈）：isBase 开启 → factor/offset 锁定 1/0；LINEAR → offset 强制 0。

### 3.4 `UnitConverter.tsx`（换算器，新增组件）

独立小组件，可作为抽屉或卡片嵌入。交互：

```
[ 物理量分类 ▼ ]  [ 源单位 ▼ ]  [ 输入值 ___ ]  →  [ 目标单位 ▼ ]  =  [ 结果 ]
                                                            换算链: 50 ℃ → 323.15 K → 122 ℉
```

- 分类联动源/目标单位下拉（`useListUnitsByCategory`）
- 输入值 `onChange` 触发 `useConvertUnit`（debounce 300ms），展示 `result` + `formula`
- `status !== CONVERT_OK` 时展示 `message`（如"对数单位不可线性换算"）

## 4. 路由（`router/modules/thingmodel.tsx`）

物模型为未来大模块，新建一级路由节点（`import.meta.glob` 自动导入）：

```tsx
import type { AppRouteObject } from '@/core/router/types';
import { createLazyRoute } from '@/core/router';

export const thingmodelRoutes: AppRouteObject[] = [
  {
    name: 'thingmodel',
    path: 'thingmodel',

      title: 'routes:thingmodel',

      {
        name: 'thingmodel-unit',
        path: 'unit',
        element: createLazyRoute(() => import('@/pages/app/thingmodel/unit')),
        meta: {
          title: 'routes:thingmodelUnit',
          icon: 'lucide:ruler',
          order: 1,
          authority: ['thingmodel:unit:view'],
        },
      },
    ],
  },
];
export default thingmodelRoutes;
```

## 5. 权限码

| 权限码 | 含义 |
|--------|------|
| `thingmodel:unit-category:view` | 查看物理量分类 |
| `thingmodel:unit-category:create` / `:update` / `:delete` | 分类增改删 |
| `thingmodel:unit:view` | 查看单位 |
| `thingmodel:unit:create` / `:update` / `:delete` | 单位增改删 |
| `thingmodel:unit:convert` | 单位换算（一般随 view 授予） |

- 路由 `meta.authority` 用 `['thingmodel:unit:view']`。
- 页面按钮用 `useAccess().hasAccessByCodes(['thingmodel:unit:create'])` 控制显隐（参照 react CLAUDE.md）。
- 超级管理员 `*:*:*` 自动放行。
- 权限码需在权限管理后台录入（permission 表 + 菜单 BUTTON 项）。

## 8. 前端注意事项（遵守 react CLAUDE.md）

1. `PaginationQuery` 必须 `new`。
2. 非组件环境（路由守卫等）只用 `fetchListUnits` / `fetchListUnitCategories`，不用 `useXxx`。
3. `DrawerForm` 用 `formRef`，无 `useForm`。
4. `ProTable` 的 `scroll.y` 初始值用 `useProTableScrollY` 返回的像素值（镜像 dict）。
5. 硬编码文本全部走 i18n。
6. `meta.title` 用 `'routes:xxx'` 格式。
7. 禁止手改 `api/generated/`。
